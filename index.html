---
Anonymous array declaration is dynamic by default.
So you can put anything in there.
---
def self.array = [
  "a", "b", "c", 6
]
    
---
Same applies to anonymous structures.
---
def self.struct = {
  a: "hello",
  b: "yolo",
    
  ---
  And this is how we define anonymous function. It is also
  dynamic by default, so all parameters are correctly typed
  by Haxe compiler
  ---
  callback: do(event)
    return "#event World"
  end
}

def self.main()
  -- Here we will call our anonymous function
  def result = struct.callback("Hello")

  -- And now, print everything to console
  trace(result)
  trace(array)
  trace(struct)

  lambdas()
end

def self.lambdas()
  def numbers = [1, 3, 5, 6, 7, 8]

  trace(numbers.count()) -- 6
  trace(numbers.has(4)) -- false

  -- test if all numbers are greater/smaller than 20
  trace(numbers.foreach(do(v)
    return v < 20
  end)) -- true

  trace(numbers.foreach(do(v)
    return v > 20
  end)) -- false

  -- sum all the numbers
  def sum = do(num, total)
    return total += num
  end

  trace(numbers.fold(sum, 0)) -- 30
end

package export;using Lambda; class Main 

{/* 
Anonymous array declaration is dynamic by default.
So you can put anything in there.
 */
static public var array = [
  "a", "b", "c", 6
];
    
/* 
Same applies to anonymous structures.
 */
static public var struct = {
  a: "hello",
  b: "yolo",
    
  /* 
  And this is how we define anonymous function. It is also
  dynamic by default, so all parameters are correctly typed
  by Haxe compiler
   */
  callback: function(event){
    return "$event World";
  }
};

static public function main(){
  // Here we will call our anonymous function
  var result = struct.callback("Hello");

  // And now, print everything to console
  trace(result);
  trace(array);
  trace(struct);

  lambdas();
}

static public function lambdas(){
  var numbers = [1, 3, 5, 6, 7, 8];

  trace(numbers.count()); // 6
  trace(numbers.has(4)); // false

  // test if all numbers are greater/smaller than 20
  trace(numbers.foreach(function(v){
    return v < 20;
  })); // true

  trace(numbers.foreach(function(v){
    return v > 20;
  })); // false

  // sum all the numbers
  var sum = function(num, total){
    return total += num;
  };

  trace(numbers.fold(sum, 0)); // 30
}}
